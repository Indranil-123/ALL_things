CLASS zcl_helloworld01 IMPLEMENTATION.

  METHOD if_oo_adt_classrun~main.

    DATA:
      lv_num1   TYPE i VALUE 20,
      lv_num2   TYPE i VALUE 10,
      lv_result TYPE i.

    out->write( '===== ARITHMETIC OPERATORS =====' ).

    " Addition
    lv_result = lv_num1 + lv_num2.
    out->write( |20 + 10 = { lv_result }| ).

    " Subtraction
    lv_result = lv_num1 - lv_num2.
    out->write( |20 - 10 = { lv_result }| ).

    " Multiplication
    lv_result = lv_num1 * lv_num2.
    out->write( |20 * 10 = { lv_result }| ).

    " Division
    lv_result = lv_num1 / lv_num2.
    out->write( |20 / 10 = { lv_result }| ).

    out->write( ' ' ).
    out->write( '===== COMPARISON OPERATORS =====' ).

    IF lv_num1 = lv_num2.
      out->write( 'Both numbers are Equal' ).
    ELSE.
      out->write( 'Both numbers are NOT Equal' ).
    ENDIF.

    IF lv_num1 > lv_num2.
      out->write( '20 is Greater than 10' ).
    ENDIF.

    IF lv_num1 < lv_num2.
      out->write( '20 is Less than 10' ).
    ELSE.
      out->write( '20 is NOT Less than 10' ).
    ENDIF.

    out->write( ' ' ).
    out->write( '===== LOGICAL OPERATORS =====' ).

    DATA:
      lv_age TYPE i VALUE 26,
      lv_salary TYPE p LENGTH 8 DECIMALS 2 VALUE '65000'.

    IF lv_age >= 18 AND lv_salary >= 50000.
      out->write( 'Eligible for Premium Employee Benefits' ).
    ENDIF.

    IF lv_age < 18 OR lv_salary < 50000.
      out->write( 'Not Eligible' ).
    ELSE.
      out->write( 'Eligibility Criteria Satisfied' ).
    ENDIF.

  ENDMETHOD.

ENDCLASS.

      FROM zemploye
      INTO TABLE @lt_employee.
